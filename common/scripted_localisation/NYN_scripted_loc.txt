defined_text = { 
	name = GetOrgDesc
	text = {
		trigger = {
			original_tag = NYN
			check_variable = {
				NYN_leftist_org > -1
				}
			NOT = {
				check_variable = {
					NYN_leftist_org > 32
					}
				}
		}
		localization_key = NYN_leftist_org_desc_low
	}
	text = {
		trigger = {
			original_tag = NYN
			check_variable = {
				NYN_leftist_org > 32
				}
			NOT = {
				check_variable = {
					NYN_leftist_org >65
					}
				}
		}
		localization_key = NYN_leftist_org_desc_med
	}
	text = {
		trigger = {
			original_tag = NYN
			check_variable = {
				NYN_leftist_org > 65
				}
		}
		localization_key = NYN_leftist_org_desc_high
	}
}
